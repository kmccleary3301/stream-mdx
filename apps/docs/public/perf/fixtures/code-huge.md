const headers = ["id", "name", "score", "status"];

function formatRow(index) {
  const name = `user_${index}`;
  const score = Math.round(Math.sin(index) * 1000) / 10;
  const status = index % 2 === 0 ? "active" : "idle";
  return { id: index, name, score, status };
}

function makeTable(rows) {
  const output = [];
  output.push(headers.join("\t"));
  for (const row of rows) {
    output.push(`${row.id}\t${row.name}\t${row.score}\t${row.status}`);
  }
  return output.join("\n");
}
